🎯 2. 핵심 탐지 알고리즘 (Python에서 검증 완료)
================================================================================

탐지 원리:
- 정상 목소리: 0-100Hz 대역 loudness ≈ 0.0000 ~ 0.0010
- 변조 목소리: 0-100Hz 대역 loudness ≈ 0.0015 ~ 0.0050
- 임계값: 0.003

판별 로직:
1. 통화 음성을 0.5초 단위로 캡처
2. FFT(Fast Fourier Transform)로 주파수 분석
3. 0-100Hz 대역 평균 loudness 계산
4. 임계값(0.003) 초과 시 → 변조 판정
5. 연속 3회 이상 탐지 시 → 알림 발생

검증 결과:
- 정확도: 100% (테스트 데이터)
- 교차 검증: 98.5%
- 실시간 처리 시간: 약 6-7ms
- 오탐률: 매우 낮음

class VoiceAnalyzer {
    private val SAMPLE_RATE = 44100  // 44.1kHz
    private val CHUNK_SIZE = 1024    // 버퍼 크기
    
    // FFT 라이브러리: JTransforms 또는 KissFFT
    
    fun analyzeLoudness(audioFrame: ShortArray): Float {
        // 1. Short → Float 변환
        val floatData = audioFrame.map { it.toFloat() / 32768f }
        
        // 2. FFT 수행
        val fftMagnitude = performFFT(floatData)
        
        // 3. 0-100Hz 인덱스 계산
        // freq = index * (SAMPLE_RATE / CHUNK_SIZE)
        // 100Hz = index * (44100 / 1024) ≈ index * 43.07
        // maxIndex = 100 / 43.07 ≈ 2.32 → 2
        val maxIndex = (100 * CHUNK_SIZE / SAMPLE_RATE).toInt()
        
        // 4. 0-100Hz 평균 계산
        var sum = 0f
        for (i in 0..maxIndex) {
            sum += fftMagnitude[i]
        }
        
        return sum / (maxIndex + 1)
    }
    
    fun isModulated(loudness: Float): Boolean {
        return loudness >= 0.003f
    }

----- CallMonitorService.kt -----

class CallMonitorService : Service() {
    private var audioRecord: AudioRecord? = null
    private val analyzer = VoiceAnalyzer()
    private val detectionBuffer = mutableListOf<Boolean>()
    
    override fun onStartCommand(intent: Intent?, flags: Int, startId: Int): Int {
        // Foreground notification
        startForeground(NOTIFICATION_ID, createNotification())
        
        // 통화 감지 리스너 등록
        val telephonyManager = getSystemService(TELEPHONY_SERVICE) as TelephonyManager
        telephonyManager.listen(phoneStateListener, PhoneStateListener.LISTEN_CALL_STATE)
        
        return START_STICKY
    }
    
    private val phoneStateListener = object : PhoneStateListener() {
        override fun onCallStateChanged(state: Int, phoneNumber: String?) {
            when (state) {
                TelephonyManager.CALL_STATE_OFFHOOK -> {
                    // 통화 시작
                    startVoiceAnalysis()
                }
                TelephonyManager.CALL_STATE_IDLE -> {
                    // 통화 종료
                    stopVoiceAnalysis()
                }
            }
        }
    }
    
    private fun startVoiceAnalysis() {
        val bufferSize = AudioRecord.getMinBufferSize(
            44100,
            AudioFormat.CHANNEL_IN_MONO,
            AudioFormat.ENCODING_PCM_16BIT
        )
        
        // VOICE_CALL 또는 VOICE_COMMUNICATION 소스 사용
        audioRecord = AudioRecord(
            MediaRecorder.AudioSource.VOICE_CALL,
            44100,
            AudioFormat.CHANNEL_IN_MONO,
            AudioFormat.ENCODING_PCM_16BIT,
            bufferSize
        )
        
        audioRecord?.startRecording()
        
        // 백그라운드 스레드에서 분석
        Thread { analyzeAudioLoop() }.start()
    }
    
    private fun analyzeAudioLoop() {
        val buffer = ShortArray(1024)
        
        while (audioRecord?.recordingState == AudioRecord.RECORDSTATE_RECORDING) {
            val read = audioRecord?.read(buffer, 0, buffer.size) ?: 0
            
            if (read > 0) {
                // 분석
                val loudness = analyzer.analyzeLoudness(buffer)
                val isModulated = analyzer.isModulated(loudness)
                
                // 연속 탐지 버퍼
                detectionBuffer.add(isModulated)
                if (detectionBuffer.size > 5) {
                    detectionBuffer.removeAt(0)
                }
                
                // 5번 중 3번 이상 탐지 → 알림
                val detectionCount = detectionBuffer.count { it }
                if (detectionCount >= 3) {
                    AlertManager.showWarning(this)
                }
            }
        }
    }
    
    private fun stopVoiceAnalysis() {
        audioRecord?.stop()
        audioRecord?.release()
        audioRecord = null
        detectionBuffer.clear()
    }
}


----- AlertManager.kt -----

object AlertManager {
    fun showWarning(context: Context) {
        // 알림 생성
        val notification = NotificationCompat.Builder(context, CHANNEL_ID)
            .setSmallIcon(R.drawable.ic_warning)
            .setContentTitle("⚠️ 보이스피싱 경고")
            .setContentText("변조된 목소리가 감지되었습니다!")
            .setPriority(NotificationCompat.PRIORITY_HIGH)
            .setCategory(NotificationCompat.CATEGORY_ALARM)
            .setVibrate(longArrayOf(0, 500, 200, 500))
            .setAutoCancel(true)
            .build()
        
        val notificationManager = context.getSystemService(NotificationManager::class.java)
        notificationManager.notify(ALERT_ID, notification)
        
        // 진동
        val vibrator = context.getSystemService(Vibrator::class.java)
        vibrator?.vibrate(VibrationEffect.createOneShot(1000, VibrationEffect.DEFAULT_AMPLITUDE))
    }
}


----- MainActivity.kt -----

class MainActivity : AppCompatActivity() {
    override fun onCreate(savedInstanceState: Bundle?) {
        super.onCreate(savedInstanceState)
        setContentView(R.layout.activity_main)
        
        // 권한 요청
        requestPermissions()
        
        // 서비스 시작 버튼
        findViewById<Button>(R.id.btnStart).setOnClickListener {
            startVoiceGuardService()
        }
        
        // 통계 표시
        updateStatistics()
    }
    
    private fun requestPermissions() {
        val permissions = arrayOf(
            Manifest.permission.READ_PHONE_STATE,
            Manifest.permission.RECORD_AUDIO,
            Manifest.permission.POST_NOTIFICATIONS
        )
        
        ActivityCompat.requestPermissions(this, permissions, REQUEST_CODE)
    }
    
    private fun startVoiceGuardService() {
        val intent = Intent(this, CallMonitorService::class.java)
        
        if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.O) {
            startForegroundService(intent)
        } else {
            startService(intent)
        }
    }
}

🚨 9. 주의사항 및 해결책
================================================================================

문제 1: Android 10+ 통화 음성 접근 제한
해결책:
- Accessibility Service 사용
- VOICE_COMMUNICATION 소스 사용
- 사용자에게 Accessibility 권한 안내

문제 2: 배터리 소모
해결책:
- 통화 중에만 작동
- 샘플링 레이트 최적화 (44100Hz → 16000Hz 고려)
- 무음 구간 스킵

문제 3: 백그라운드 실행 제한
해결책:
- Foreground Service 사용 (필수)
- 알림 항상 표시
- 배터리 최적화 제외 요청

문제 4: 오탐률
해결책:
- 연속 3회 이상 탐지 시에만 알림
- 임계값 조정 가능하게 (설정)
- 사용자 피드백 수집

----- 개선된 탐지 알고리즘 -----

특징:
1. 8개 특징 사용 (0-100Hz, 100-200Hz, 200-300Hz, 4-8kHz, 16kHz+, 
                 Total Energy, Spectral Centroid, ZCR)
2. AI 모델 신뢰도 임계값: 0.80 (80% 이상 확신할 때만)
3. 특징 기반 검증:
   - 0-100Hz >= 0.002 또는
   - 100-200Hz >= 0.0035
4. 연속 탐지: 최근 5번 중 3번 이상 탐지되어야 알림
5. 오탐 방지 카운터 추적


----- Kotlin 구현 (개선 버전) -----

class ImprovedVoiceAnalyzer {
    // 임계값 설정
    private val CONFIDENCE_THRESHOLD = 0.80f
    private val LOUDNESS_0_100_MIN = 0.0020f
    private val LOUDNESS_100_200_MIN = 0.0035f
    private val CONSECUTIVE_THRESHOLD = 3
    
    // 연속 탐지 버퍼
    private val predictionBuffer = ArrayDeque<PredictionResult>(5)
    
    // 오탐 방지 카운터
    private var falsePositivePrevention = 0
    
    data class AudioFeatures(
        val loudness0to100: Float,
        val loudness100to200: Float,
        val loudness200to300: Float,
        val loudness4to8k: Float,
        val loudness16kPlus: Float,
        val totalEnergy: Float,
        val spectralCentroid: Float,
        val zcr: Float
    )
    
    data class PredictionResult(
        val prediction: Int,  // 0=정상, 1=변조
        val probability: Float,
        val loudness0to100: Float,
        val loudness100to200: Float
    )
    
    data class DetectionResult(
        val isModulated: Boolean,
        val confidence: Float,
        val aiConfidence: Float,
        val hasSignature: Boolean,
        val bufferCount: String,
        val falsePositivePrevented: Int
    )
    
    fun extractFeatures(audioFrame: ShortArray): AudioFeatures {
        // FFT 수행
        val fftMagnitude = performFFT(audioFrame)
        val freqs = getFrequencies(audioFrame.size)
        
        return AudioFeatures(
            loudness0to100 = calculateBandLoudness(fftMagnitude, freqs, 0f, 100f),
            loudness100to200 = calculateBandLoudness(fftMagnitude, freqs, 100f, 200f),
            loudness200to300 = calculateBandLoudness(fftMagnitude, freqs, 200f, 300f),
            loudness4to8k = calculateBandLoudness(fftMagnitude, freqs, 4000f, 8000f),
            loudness16kPlus = calculateBandLoudness(fftMagnitude, freqs, 16000f, 22050f),
            totalEnergy = fftMagnitude.sumOf { (it * it).toDouble() }.toFloat(),
            spectralCentroid = calculateSpectralCentroid(fftMagnitude, freqs),
            zcr = calculateZCR(audioFrame)
        )
    }
    
    fun predictImproved(audioFrame: ShortArray, aiModel: AIModel?): DetectionResult {
        // 1. 특징 추출
        val features = extractFeatures(audioFrame)
        
        // 2. AI 예측 (모델이 있으면)
        val aiPrediction: Int
        val aiProbability: Float
        
        if (aiModel != null) {
            val (pred, prob) = aiModel.predict(features)
            aiPrediction = pred
            aiProbability = prob
        } else {
            // AI 모델 없으면 규칙 기반
            aiPrediction = if (features.loudness0to100 >= 0.003f) 1 else 0
            aiProbability = if (aiPrediction == 1) 0.9f else 0.1f
        }
        
        // 3. 특징 기반 검증
        val hasSignature = (
            features.loudness0to100 >= LOUDNESS_0_100_MIN ||
            features.loudness100to200 >= LOUDNESS_100_200_MIN
        )
        
        // 4. AI + 규칙 결합
        val framePrediction: Int = if (aiPrediction == 1) {
            if (aiProbability >= CONFIDENCE_THRESHOLD && hasSignature) {
                // 신뢰도 높고 + 명확한 특징 → 변조
                1
            } else {
                // 조건 불충분 → 정상으로 수정
                falsePositivePrevention++
                0
            }
        } else {
            0
        }
        
        // 5. 버퍼에 추가
        predictionBuffer.addLast(
            PredictionResult(
                prediction = framePrediction,
                probability = aiProbability,
                loudness0to100 = features.loudness0to100,
                loudness100to200 = features.loudness100to200
            )
        )
        
        if (predictionBuffer.size > 5) {
            predictionBuffer.removeFirst()
        }
        
        // 6. 연속 탐지 (스무딩)
        val isModulated: Boolean
        val avgProbability: Float
        
        if (predictionBuffer.size >= 3) {
            val modulatedCount = predictionBuffer.count { it.prediction == 1 }
            isModulated = modulatedCount >= CONSECUTIVE_THRESHOLD
            avgProbability = predictionBuffer.map { it.probability }.average().toFloat()
        } else {
            isModulated = false
            avgProbability = 0.5f
        }
        
        return DetectionResult(
            isModulated = isModulated,
            confidence = if (isModulated) avgProbability else 1 - avgProbability,
            aiConfidence = aiProbability,
            hasSignature = hasSignature,
            bufferCount = "${predictionBuffer.count { it.prediction == 1 }}/${predictionBuffer.size}",
            falsePositivePrevented = falsePositivePrevention
        )
    }
    
    private fun calculateBandLoudness(
        fftMagnitude: FloatArray,
        freqs: FloatArray,
        minFreq: Float,
        maxFreq: Float
    ): Float {
        var sum = 0f
        var count = 0
        
        for (i in fftMagnitude.indices) {
            if (freqs[i] in minFreq..maxFreq) {
                sum += fftMagnitude[i]
                count++
            }
        }
        
        return if (count > 0) sum / count else 0f
    }
    
    private fun calculateSpectralCentroid(
        fftMagnitude: FloatArray,
        freqs: FloatArray
    ): Float {
        var numerator = 0f
        var denominator = 0f
        
        for (i in fftMagnitude.indices) {
            numerator += freqs[i] * fftMagnitude[i]
            denominator += fftMagnitude[i]
        }
        
        return if (denominator > 0) numerator / denominator else 0f
    }
    
    private fun calculateZCR(audioFrame: ShortArray): Float {
        var count = 0
        for (i in 1 until audioFrame.size) {
            if ((audioFrame[i] >= 0 && audioFrame[i-1] < 0) ||
                (audioFrame[i] < 0 && audioFrame[i-1] >= 0)) {
                count++
            }
        }
        return count.toFloat() / (2 * audioFrame.size)
    }
}

class CallMonitorService : Service() {
    private val analyzer = ImprovedVoiceAnalyzer()
    private var aiModel: AIModel? = null  // TFLite 모델 (선택)
    
    private fun analyzeAudioLoop() {
        val buffer = ShortArray(1024)
        
        while (audioRecord?.recordingState == AudioRecord.RECORDSTATE_RECORDING) {
            val read = audioRecord?.read(buffer, 0, buffer.size) ?: 0
            
            if (read > 0) {
                // 개선된 분석
                val result = analyzer.predictImproved(buffer, aiModel)
                
                // 변조 탐지 시
                if (result.isModulated) {
                    AlertManager.showWarning(
                        context = this,
                        confidence = result.confidence,
                        details = "신뢰도: ${(result.confidence * 100).toInt()}%\n" +
                                 "AI 확률: ${(result.aiConfidence * 100).toInt()}%\n" +
                                 "버퍼: ${result.bufferCount}\n" +
                                 "특징 검출: ${if (result.hasSignature) "있음" else "없음"}"
                    )
                }
                
                // 로그 (디버그용)
                Log.d(TAG, "변조: ${result.isModulated}, " +
                           "신뢰도: ${result.confidence}, " +
                           "오탐방지: ${result.falsePositivePrevented}")
            }
        }
    }
}

----- AIModel.kt (TFLite 래퍼) -----

class AIModel(context: Context) {
    private val interpreter: Interpreter
    
    init {
        // assets에서 모델 로드
        val modelFile = loadModelFile(context, "voice_classifier.tflite")
        interpreter = Interpreter(modelFile)
    }
    
    fun predict(features: ImprovedVoiceAnalyzer.AudioFeatures): Pair<Int, Float> {
        // 입력 배열 (8개 특징)
        val input = floatArrayOf(
            features.loudness0to100,
            features.loudness100to200,
            features.loudness200to300,
            features.loudness4to8k,
            features.loudness16kPlus,
            features.totalEnergy,
            features.spectralCentroid,
            features.zcr
        )
        
        // 출력 배열 [정상 확률, 변조 확률]
        val output = Array(1) { FloatArray(2) }
        
        // 예측
        interpreter.run(input, output)
        
        val normalProb = output[0][0]
        val modulatedProb = output[0][1]
        
        val prediction = if (modulatedProb > normalProb) 1 else 0
        val confidence = maxOf(normalProb, modulatedProb)
        
        return Pair(prediction, confidence)
    }
    
    private fun loadModelFile(context: Context, filename: String): ByteBuffer {
        val assetFileDescriptor = context.assets.openFd(filename)
        val inputStream = FileInputStream(assetFileDescriptor.fileDescriptor)
        val fileChannel = inputStream.channel
        val startOffset = assetFileDescriptor.startOffset
        val declaredLength = assetFileDescriptor.declaredLength
        
        return fileChannel.map(FileChannel.MapMode.READ_ONLY, startOffset, declaredLength)
    }
    
    fun close() {
        interpreter.close()
    }
}
